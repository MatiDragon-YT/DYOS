Menu_Display_Hide()
repeat
	wait 0
	
	if CURRENT_PLATFORM == __PLATFORM_WINDOWS
	then get_cleo_shared_var 4@ = CAMARA_AGITACION
	else Android.GetVar(4@, CAMARA_AGITACION)
	end

	if VK_Left()
	then
		4@ -= 1
		if 4@ < 0
		then 4@ = 0
		end
	end
	if VK_Right()
	then
		4@ += 1
		if 4@ > 1000
		then 4@ = 1000
		end
	end
	if ParaAtras()
	then
		4@ = 0
	end
	Camera.Shake(4@)
	Text.PrintWithNumberNow('M0000JT', 4@, 10, 0)

	if CURRENT_PLATFORM == __PLATFORM_WINDOWS
	then set_cleo_shared_var CAMARA_AGITACION = 4@
	else Android.SetVar(CAMARA_AGITACION, 4@)
	end

until Menu_Return()
Menu_Display_Show()
Menu.SetActiveItem(hPanel, 5)
jump @MenuDeLaCamara
