Menu_Display_Hide()
repeat
	wait 0
	
	if $ModoDeCompilacion == __PLATFORM_WINDOWS
	then get_cleo_shared_var 4@ = CAMARA_HEBRIEDAD
	else Android.GetVar(4@, CAMARA_HEBRIEDAD)
	end

	if VK_Left()
	then
		4@ -= 1
		if 4@ < 0
		then 4@ = 0
		end
	end
	if VK_Right()
	then
		4@ += 1
		if 4@ > 255
		then 4@ = 255
		end
	end
	if ParaAtras()
	then
		4@ = 0
	end
	Player.SetDrunkVisuals(hChar, 4@)
	Text.PrintWithNumberNow('M0000JT', 4@, 10, 0)

	if $ModoDeCompilacion == __PLATFORM_WINDOWS
	then set_cleo_shared_var CAMARA_HEBRIEDAD = 4@
	else Android.SetVar(CAMARA_HEBRIEDAD, 4@)
	end
until Menu_Return()
Menu_Display_Show()
Menu.SetActiveItem(hPanel, 6)
jump @MenuDeLaCamara
