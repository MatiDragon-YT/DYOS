Menu.SetActiveItem(hPanel, 0)
:MenuDeLaPosicionDelJugador

Menu.SetColumn(hPanel, 0, 'M00000E','M0000RW','M000000','M0000FU','M0000MM','M0000NL','M00003M','M0000FP','M0000EO','M0000EP','M000001',none,none) 


int 0@ = B_VAR_PLAYER_PROPS
bget_var()
int 1@ = 0@

0@ = 0
if lbit_is_false 1@ PLAYER_PROP_COLISION // is FALSE
then setMenuFalse()
else setMenuTrue()
end

Menu_Take_Return() 
Menu_Take_Selection()
repeat
	wait 0
    if Menu_Close_Forced()
    then jump @Menu_Close
    end
	if Menu_Enter()
	then 
		if _SELECTED == 0 // Solid
		then
			int 0@ = B_VAR_PLAYER_PROPS
			bget_var()
			int 1@ = 0@
			0@ = 0
			if lbit_is_false 1@ PLAYER_PROP_COLISION
			then
				lbit_set_true 1@ PLAYER_PROP_COLISION
				Actor.SetKeepPosition(hPlayer) = false
				Actor.SetCollisionDetection(hPlayer) = false
				Enable()
				setMenuTrue()
			else
				lbit_set_false 1@ PLAYER_PROP_COLISION
				Actor.SetKeepPosition(hPlayer) = true
				Actor.SetCollisionDetection(hPlayer) = true
				Disable()
				setMenuFalse()
			end
			call @bset_var 2 B_VAR_PLAYER_PROPS = 1@
		end

		if _SELECTED == 1 // Free movement
		then
			:movimiento_libre_del_jugador
			{$I ./location/free.txt}
			Menu.SetActiveItem(hPanel,1)
			jump @MenuDeLaPosicionDelJugador
		end

		if _SELECTED == 2 // Dentro del veh.
		then
			{$I ./location/enter_car.txt}
			Menu.SetActiveItem(hPanel,2)
			jump @MenuDeLaPosicionDelJugador
		end

		if _SELECTED == 3 // Ancla. al vehiculo
		then
			{$I ./location/attach_car.txt}
			Menu.SetActiveItem(hPanel,3)
			jump @MenuDeLaPosicionDelJugador
		end
		
		if _SELECTED == 4 // Ancla. al objecto
		then
			{$I ./location/attach_object.txt}
			Menu.SetActiveItem(hPanel,4)
			jump @MenuDeLaPosicionDelJugador
		end

		if _SELECTED == 5 // TP to interior
		then
			{$I ./location/tp_interior.txt}
			Menu.SetActiveItem(hPanel, 5)
			jump @MenuDeLaPosicionDelJugador
		end

		if _SELECTED == 6 // TP to actor
		then
			{$I ./location/tp_actor.txt}
			Menu.SetActiveItem(hPanel, 6)
			jump @MenuDeLaPosicionDelJugador
		end

		if _SELECTED == 7 // TP to vehicle
		then
			{$I ./location/tp_car.txt}
			Menu.SetActiveItem(hPanel, 7)
			jump @MenuDeLaPosicionDelJugador
		end

		if _SELECTED == 8 // TP to object
		then
			{$I ./location/tp_object.txt}
			Menu.SetActiveItem(hPanel, 8)
			jump @MenuDeLaPosicionDelJugador
		end

		if _SELECTED == 9 // TP to marker
		then
			{$I ./location/tp_marker.txt}
		end

		GetUpdatePlayerPosition()
		Menu_Take_Selection()
	end
until Menu_Return()
setMenuClear()
