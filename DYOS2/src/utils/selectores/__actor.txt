:Actor_GetCreatedWithDYOS
	int 0@ = B_VAR_LAST_ACTOR_SELECTED
	bget_var()
	int $indActor = 0@

	if $numActores >= 2
	then
		if $ModoDeCompilacion == f_PC
		then Text.PrintHelp('M0000JK')
		else Text.PrintHelp('M0000JL')
		end
		Menu_Display_Hide()
		Menu_Take_Selection()
		repeat
			Wait0()
			Actor_GetDataAtDisplay()
			iluminar_al_actor()
			Text.PrintWith2NumbersNow('M0000BG',$indActor,$numActores,150,0)

			if Blip.Defined($ResaltadorDeEntidad)
			then
				Wait0()
				Actor_GetDataAtDisplay()
				iluminar_al_actor()
			else
				$ResaltadorDeEntidad = Blip.CreateAboveActor($Actor($indActor,255i))
				while VK_Left()
					Wait0()
					Actor_GetDataAtDisplay()
					iluminar_al_actor()
					FixUpdate()
				end
				while VK_Right()
					Wait0()
					Actor_GetDataAtDisplay()
					iluminar_al_actor()
					FixUpdate()
				end
			end

			if VK_Left()
			then
				ActorDefinidoInferior()
				Blip.Remove($ResaltadorDeEntidad)
			end
			if VK_Right()
			then
				ActorDefinidoSuperior()
				Blip.Remove($ResaltadorDeEntidad)
			end

			if Menu_Select()
			then
				Menu_Display_Show()
				Menu.SetActiveItem($PANEL, 0)
				Text.ClearHelp()
				Text.ClearPrints()
				break
			end
		until Menu_Return()
		Blip.Remove($ResaltadorDeEntidad)
		Menu_Display_Show()
		Text.ClearHelp()
		Text.ClearPrints()
	end
return
