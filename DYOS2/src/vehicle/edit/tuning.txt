		then
			0@ = Car.GetModel($Auto($indAuto,255i))
			if Model.IsCar(0@)
			then
				// [...]
			else
				Text.PrintHelp('M0000OH')
			end
		end

		then
			0@ = Car.GetModel($Auto($indAuto,255i))
			if Model.IsCar(0@)
			then

				if _SELECTED == 2 // Alerones

				Text.PrintHelp('M0000OH')

				Disable()
