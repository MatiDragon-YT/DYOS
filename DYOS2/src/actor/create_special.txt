	:f_create_special_actor
		if Menu_Enter()
		then
			if $numActores < 50
			then
				int 0@ = 1
				while 0@ <= $numActores
					$indActor = 0@
					if Actor.Defined($Actor($indActor,255i))
					then
						0@ += 1
					else
						0@ *= -1
						break
					end
				end
				if 0@ < 0
				then 0@ *= -1
				else
					$indActor = $numActores
					$numActores++
					$indActor++
				end

				15@ += _SELECTED


				LoadSpecialActor()
				repeat
					wait 0
					FixUpdate()
				until SpecialActor.Available(1)
				
				GetUpdatePlayerPosition()
				$Actor($indActor,255i) = Actor.Create(PedType.Special, #SPECIAL01, $tempvar_X_coord, $tempvar_Y_coord, $tempvar_Z_coord)
				Actor.Angle($Actor($indActor,255i)) = $tempvar_Angle
				Actor.SetLinkInInterior($Actor($indActor,255i)) = $Active_Interior
				SpecialActor.Release(1)
				Actor_ResetProperties()
				int 0@ = B_ACTOR_TYPE
				int 1@ = $indActor
				int 2@ = 23
				bset_actor()
				int 0@ = B_ACTOR_GENERATION
				int 1@ = $indActor
				int 2@ = GENERATION_SPECIAL
				bset_actor()
				int 0@ = B_ACTOR_MODEL_SPECIAL
				int 1@ = $indActor
				int 2@ = 15@
				bset_actor()
				Menu.SetActiveItem(hPanel, 0)
				return_true
			end
		else return_false
		end
	return
	
	:f_player_set_model_special_actor
		if ParaEspecial()
		then
			15@ += _SELECTED
			LoadSpecialActor()
			Text.PrintHelpForever('M0000AW')
			repeat
				wait 0
			until SpecialActor.Available(1)
			Text.ClearHelp()
			Player.SetModel(hChar, #SPECIAL01)
			SpecialActor.Release(1)
			Menu_Take_Selection()
		end
	return
