:CrearObjeto 
	Menu_Take_Selection()
	call @bget_var4 1 B_VAR_MODEL_OBJECT = 6@
	$indObjeto = $numObjetos
	
	19@ = 0
	20@ = 0
	if $numObjetos < 50
	then
		int 0@ = 1
		while 0@ <= $numObjetos
			$indObjeto = 0@
			if Object.Defined($Objeto($indObjeto,255i))
			then
				0@ += 1
			else
				0@ *= -1
				break
			end
		end
		if 0@ < 0
		then 0@ *= -1
		else
			$indObjeto = $numObjetos
			$numObjetos++
			$indObjeto++
		end
		GenerarObjeto()
		Menu_Display_Hide()
		25@ = 2
		repeat
			wait 0
			FixUpdate()
			
			if ParaEspecial()
			then
				if 19@ == 0
				then 19@ = 1
				else 19@ = 0
				end
				FrenarEspecial()
			end

			if 19@ == 1
			then InputNumberAndroid()
			else
				if $ModoDeCompilacion == __PLATFORM_WINDOWS
				then InputNumberPC()
				end
				Text.PrintWithNumberNow('M000007', 6@, 200, 0)

				if VK_Left()
				then
					Model.Release(6@)
					6@--
					ModeloAnterior_Objeto()
				end
				if VK_Right()
				then
					Model.Release(6@)
					6@++
					ModeloSiguiente_Objeto()
				end
				if ParaAtras()
				then
					Model.Release(6@)
					6@ -= 10
					ModeloAnterior_Objeto()
				end
				if ParaAdelante()
				then
					Model.Release(6@)
					6@ += 10
					ModeloSiguiente_Objeto()
				end
				if ParaDisminuir()
				then
					Model.Release(6@)
					6@ -= 100
					ModeloAnterior_Objeto()
				end
				if ParaAumentar()
				then
					Model.Release(6@)
					6@ += 100
					ModeloSiguiente_Objeto()
				end			
			end
			
			param1r = false
			if Menu_Select()
			then
				param1r = true
				if not Object.Model($Objeto($indObjeto,255i)) == 6@
				then
					Object.Remove($Objeto($indObjeto,255i))
					GenerarObjeto()
				end

				if bit_run $statsObjet($indObjeto,255i) OBJETO_PROP_HAVE_PARTICLE
				then
					int 10@
					int 6@
					
					0@ = PARTICLE_HANDLE
					int 1@ = $indObjeto
					int 2@ = 10@
					bset_particle()

					0@ = PARTICLE_ID
					int 1@ = $indObjeto
					int 2@ = 6@
					bset_particle()
				end
				break
			end
			if ParaAbajo()
			then
				param1r = undefined
				break
			end

		until Menu_Return()


		Menu_Display_Show()
		Menu.SetActiveItem(hPanel, 0)
		Text.ClearHelp()
		Text.ClearPrints()
	else
		Text.PrintHelp('M00000M')
	end
return

	bit_set_false $statsObjet($indObjeto,255i) OBJETO_PROP_MOVIBLE

	if 6@ > 0
	then 
		Model.Load(6@)
		Text.PrintHelpForever('M0000AW')
		if $ModoDeCompilacion == __PLATFORM_WINDOWS
		then
			TIMERB = 0
			while TIMERB <= 84
				Wait0()
			end
		else
			TIMERZ = 0
			while TIMERZ <= 84
				Wait0()
			end
		end
		repeat
			Wait0()
		until Model.Available(6@)
		Text.ClearHelp()
		
		GetUpdatePlayerPosition()
		$Objeto($indObjeto,255i) = Object.Create(6@, $tempvar_X_coord, $tempvar_Y_coord, $tempvar_Z_coord)
		Object.Angle($Objeto($indObjeto,255i)) = $tempvar_Angle
		Object.SetLinkInInterior($Objeto($indObjeto,255i)) = $Active_Interior
		Object.SetCollisionDetection($Objeto($indObjeto,255i)) = false
		Object.SetDestructible($Objeto($indObjeto,255i)) = false

		Model.Release(6@)

		bit_set_false $statsObjet($indObjeto,255i) OBJETO_PROP_HAVE_PARTICLE
	else
		Text.PrintHelpForever('M0000AW')
		Model.Load(3106)
		if $ModoDeCompilacion == __PLATFORM_WINDOWS
		then
			TIMERB = 0
			while TIMERB <= 84
				Wait0()
			end
		else
			TIMERZ = 0
			while TIMERZ <= 84
				Wait0()
			end
		end
		repeat
			Wait0()
		until Model.Available(3106)
		Text.ClearHelp()
		
		GetUpdatePlayerPosition()
		float 8@ = $tempvar_Z_coord
		8@ += 1.0
		$Objeto($indObjeto,255i) = Object.Create(3106, $tempvar_X_coord, $tempvar_Y_coord, 8@)
		Object.Angle($Objeto($indObjeto,255i)) = $tempvar_Angle
		Object.SetLinkInInterior($Objeto($indObjeto,255i)) = $Active_Interior
		Object.SetCollisionDetection($Objeto($indObjeto,255i)) = false
		Object.SetDestructible($Objeto($indObjeto,255i)) = false
		
		Model.Release(3106)

		bit_set_true $statsObjet($indObjeto,255i) OBJETO_PROP_HAVE_PARTICLE

		ParticleIDs()
		Particle.Play(10@)
	end

	int 0@ = B_OBJETO_ATTACHED_TO
	int 1@ = $indObjeto
	float 2@ = ATTACHED_TO_NONE
	bset_object()

	int 0@ = B_OBJETO_LCOORD_X
	int 1@ = $indObjeto
	float 2@ = 0.0
	bset_object()

	int 0@ = B_OBJETO_LCOORD_Y
	int 1@ = $indObjeto
	float 2@ = 0.0
	bset_object()

	int 0@ = B_OBJETO_LCOORD_Z
	int 1@ = $indObjeto
	float 2@ = 0.0
	bset_object()

	int 0@ = B_OBJETO_LROT_X
	int 1@ = $indObjeto
	float 2@ = 0.0
	bset_object()

	int 0@ = B_OBJETO_LROT_Y
	int 1@ = $indObjeto
	float 2@ = 0.0
	bset_object()

	int 0@ = B_OBJETO_LROT_Z
	int 1@ = $indObjeto
	float 2@ = 0.0
	bset_object()
return
