if CURRENT_PLATFORM == __PLATFORM_WINDOWS
then Text.PrintHelp('M0000JK') // ~p~TAB~s~: toma el mas cercano.~n~~p~NUMERO~s~: toma el que marques.
else Text.PrintHelp('M0000JL') // ~p~DEAL~s~: toma el mas cercano.~n~~p~HOLD~s~: toma el que marques.
end

Menu_Display_Hide()
Menu_Take_Selection()

Actor.GetPosition(SM_ACTOR, 1@, 2@, 3@)
Actor.GetPosition(hPlayer, 4@, 5@, 6@)
050A: 21@ = distance_between_XYZ 4@ 5@ 6@ and_XYZ 1@ 2@ 3@

call @SM_Get_Total_Actor 0 20@
repeat
	//Text.PrintWith2NumbersNow('M0000BG',SM_INDEX_ACTORS,SM_NUM_ACTORS,150,0) // __~1~/~1~__

	printf "[%i/%i] MTS(%.2f) ~n~R(%i)" 10 SM_INDEX_ACTORS SM_NUM_ACTORS 21@ 20@

	if VK_Tab()
	then
		SM_INDEX_ACTORS = 0
		10@ = 0.1
		12@ = false

		while true 
			Text.Print('M0000MX',10,0) // Buscando...
			12@ = true
			SM_INDEX_ACTORS++
			
			if SM_INDEX_ACTORS > SM_NUM_ACTORS
			then
				SM_INDEX_ACTORS = 1
				10@ += 0.1
				if 10@ > 75.0
				then
					Text.ClearHelp()
					Text.PrintHelp('M0000L6') // Acerquese mas a uno de los elementos.
					12@ = false
					break
				end
			end
			
			call @Array_Actor_Get 2 SM_INDEX_ACTORS B_ACTOR_HANDLE value SM_ACTOR
			if Actor.Defined(SM_ACTOR)
			then
				if Actor.IsLocatedNearActor3D(hPlayer,0,SM_ACTOR,10@,10@,10@)
				then break
				end
			end
		end
		Blip.Remove(hEntityHighlighter)
		if 12@ == true
		then
			Menu_Display_Show()
			EXIST_ELEMENT_ILUMINATED = ELEMENT_ILUMINATED_NONE
			Menu.SetActiveItem(hPanel, 0)
			Text.ClearHelp()
			Text.ClearPrints()
			jump @MenuDeEdicionDelActor
		end
	end

	if not Blip.Defined(hEntityHighlighter)
	then
		hEntityHighlighter = Blip.CreateAboveActor(SM_ACTOR)
		Actor.GetPosition(SM_ACTOR, 1@, 2@, 3@)
		Actor.GetPosition(hPlayer, 4@, 5@, 6@)
		050A: 21@ = distance_between_XYZ 4@ 5@ 6@ and_XYZ 1@ 2@ 3@
	else
		//Wait0()
	end
			

	if VK_Left_Menu()
	then
		ActorDefinidoInferior()
		Blip.Remove(hEntityHighlighter)
	end
	if VK_Right_Menu()
	then
		ActorDefinidoSuperior()
		Blip.Remove(hEntityHighlighter)
	end

	if Menu_Select()
	then
		Menu_Display_Show()
		ACTOR_LAST_SELECTED = SM_INDEX_ACTORS
		Menu.SetActiveItem(hPanel, 0)
		Text.ClearHelp()
		Text.ClearPrints()
		jump @MenuDeEdicionDelActor
	end
until Menu_Return()
Blip.Remove(hEntityHighlighter)
Menu_Display_Show()
