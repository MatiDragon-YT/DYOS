Menu.SetActiveItem(hPanel, 0)

:MenuQueActorCrear

NUMBER_PAGE = 1
CURRENT_PAGE = 1
Menu.SetColumn(hPanel,0,'M0000DO','M0000AH','M0000IQ','M0000MU','M0000WO',none,none,none,none,none,none,none,none)

if SCM_NUM_ACTORS >= 1
then Menu.SetItemWithNumber(hPanel, 0, 4, 'M0000TW', 0)
end

Menu_Take_Return()
Menu_Take_Selection()
repeat
	Wait0()
	if Menu_Close_Forced()
	then jump @Menu_Close
	end
	if Menu_Enter()
	then
		if _SELECTED == 0 // Ped
		then
			{$I ./create/ped.txt}
			
			if param1r == true
			then
				if 6@ >= 0
				then
					call @bset_var 2 B_VAR_MODEL_ACTOR = 6@
				end
				call @bset_var 2 B_VAR_LAST_ACTOR_SELECTED = SCM_INDEX_ACTORS
				jump @MenuDeEdicionDelActor
			end
			if param1r == false
			then 
				SCM_NUM_ACTORS--
				Actor.Remove($Actor(SCM_INDEX_ACTORS,255i))
				SCM_INDEX_ACTORS--
				jump @MenuQueActorCrear
			end
			if param1r == undefined
			then jump @MenuQueActorCrear
			end
		end

	end
until Menu_Return()
