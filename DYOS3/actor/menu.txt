  // for debug:

	if Menu_Close_Forced()
	then jump @Menu_Close
	end

	call @MiniSCM_ActionsAvailables 1 shows 7
	if VK_Ctrl_PressedOnce()
	  then
	    if 0A9A: FILE_HANDLE = DIR_DYOS "ab"
		  then
				switch _SELECTED
			    case 0 // Create
	          FILE_COMMAND = VM_ACTOR_CREATE
	          scm_write_command()
	    		default
			      nop
			  end
			  Fs.Close(FILE_HANDLE)
	    end
	  end

	if Menu_Enter()
	then

		if _SELECTED == 0 // Create
		then
			Menu.SetActiveItem(hPanel, 0)
			{$I ./create.txt}

			jump @Menu_iActor

			{$I ./edit.txt}

		end

		if _SELECTED == 3 // Select
		then
			{$I ./select.txt}
			Menu.SetActiveItem(hPanel, 3)
			jump @Menu_iActor
		end

	end
until Menu_Return()
