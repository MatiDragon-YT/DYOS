:SMX_read_command
// FILE_VERSION [XXX][X][XX]
//
// [XXX] : INT version (distribution.version.parch)
// [X] : BOOL (0-alpha or 1-beta)
// [XX] : prerelease

if FILE_COMMAND >= 0x8000
then SMX_IF_ACTIVE = 1
end

switch FILE_COMMAND
// AREAS IDs
// 0x0000 - 0x0FFF : Command simple
// 0x8000 - 0x8FFF : Command conditional
// 0x0000 : Control flow

case VM_WAIT

    SMX_read_int()
    call @waitPlis 2 SMX_A false

case VM_GOTO

    SMX_read_int()
    SMX_CURRENT_OFFSET = SMX_A

case VM_IF

    SMX_read_int()
    SMX_IF_TYPE = SMX_A        // 0 = AND, 1 = OR
    SMX_read_int()
    SMX_IF_FALSE_JUMP = SMX_A  // false jump

    // Resolver IF
    if SMX_IF_ACTIVE == 1
    then
        if SMX_IF_TYPE == 0 // AND
        then
            if SMX_CONT_CONDITION <> SMX_IF_TRUE
            then
                SMX_CURRENT_OFFSET = SMX_IF_FALSE_JUMP
            end
        else // OR
            if SMX_CONT_CONDITION == 0
            then
                SMX_CURRENT_OFFSET = SMX_IF_FALSE_JUMP
            end
        end
        
        if Actor.IsDead(hPlayer)
        then
            SMX_MISSION_FAIL = true

            printf "SMX_MISSION_FAIL" 2000
        end
    end

    SMX_IF_ACTIVE = 0
    SMX_IF_TRUE = 0
    SMX_CONT_CONDITION = 0

    if SMX_ACTOR_TO_KILL > 0
    then printf "sleep to %i actors" 84 SMX_ACTOR_TO_KILL
    end
    SMX_ACTOR_TO_KILL = 0
    :SMX_goto_mission_fake

case VM_END_THREAD

    SMX_CURRENT_OFFSET = FILE_SIZE

case VM_NOP

    nop

//   0x1000 : Player

case VM_PLAYER_SET_POSITION

    SMX_read_3_float()
    Actor.SetPosition(hPlayer,SMX_Af,SMX_Bf,SMX_Cf)

case VM_PLAYER_SET_WANTED

    SMX_read_int()
    Player.SetWantedLevel(hChar) = SMX_A

case VM_PLAYER_SET_WEAPON

    SMX_read_int()
    //0A9D: read_file FILE_HANDLE size 4 to hPlayer // Actor
    SMX_read_3_int()
    call @GiveWeaponAndAmmo 4 actor hPlayer model_id SMX_A weapon_id SMX_B ammo SMX_C

case VM_PLAYER_SET_ANIM

    SMX_read_3_int()

    CURRENT_ANIM_ID = SMX_B
    if SMX_A == PLAYER_ANIM_LIB_DYOM
    then AnimDyom()
    end
    if SMX_A == PLAYER_ANIM_LIB_DYOS
    then AnimDyos()
    end
    
    if ANIM_NOT_IS_OF_PED == false
    then
        if lbit_run SMX_C PLAYER_PROP_COLISION
        then Task.PlayAnimSecondary(hPlayer, IFP_ANIM, "PED", 30.0, true, false, false, false, -1)
        else Task.PlayAnim(hPlayer, IFP_ANIM, "PED", 30.0, true, false, false, false, -1)
        end
    else
        Animation.Load(IFP_FILE)
        repeat
            Wait0()
        until Animation.Available(IFP_FILE)
        if lbit_run SMX_C PLAYER_PROP_COLISION
        then Task.PlayAnimSecondary(hPlayer, IFP_ANIM, IFP_FILE, 4.0, true, false, false, false, -1)
        else Task.PlayAnim(hPlayer, IFP_ANIM, IFP_FILE, 4.0, true, false, false, false, -1)
        end
        Animation.Remove(IFP_FILE)
    end

case VM_PLAYER_SET_FIGHT

    SMX_read_int()
    Actor.SetFightStyle(hPlayer, SMX_A, 6)
    
case VM_PLAYER_SET_HEALTH

    SMX_read_int()
    Actor.SetHealth(hPlayer) = SMX_A

//   0x2000 : Actor/Task

case VM_ACTOR_CREATE

    SMX_read_int() // id
    // Por medidas de seguridad, vemos si el ID no existe ya en el buffer
    call @SMX_Get_Index_Actor 1 SMX_A result SMX_INDEX_ACTORS
    // Si tenemos un indice, solo salteamos todas las lecturas
    if SMX_INDEX_ACTORS <> -1
    then
        SMX_A = 8
        SMX_A *= 4 // 8 (type, model, vec3, angle, flags, gun) * 4
        SMX_CURRENT_OFFSET += SMX_A
    else
        SMX_NUM_ACTORS += 1

        call @SMX_Array_Actor_Set 3 SMX_NUM_ACTORS B_ACTOR_ID value SMX_A

        SMX_read_int() // type
        SMX_G = SMX_A
        SMX_read_int() // model
        ACTOR_MODEL_REQ = SMX_A

        SMX_read_3_float() // coords
        SMX_Hf = SMX_Af
        SMX_If = SMX_Bf
        SMX_Jf = SMX_Cf
        SMX_read_float()  // angle
        SMX_Kf = SMX_Af

        0@ = undefined
        CargarActorModelo()

        0980: extinguish_all_fires_at SMX_Hf SMX_If SMX_Jf 1.0

        if and
            ACTOR_MODEL_REQ >= 0
            ACTOR_MODEL_REQ < 1000
        then
            Actor.Create(SMX_ACTOR, SMX_G, ACTOR_MODEL_REQ, SMX_Hf, SMX_If, SMX_Jf)
            Model.Release(ACTOR_MODEL_REQ)
        else
            Actor.Create(SMX_ACTOR, PedType.Special, #SPECIAL01, SMX_Hf, SMX_If, SMX_Jf)
            SpecialActor.Release(1)
        end
        call @SMX_Array_Actor_Set 3 SMX_NUM_ACTORS B_ACTOR_HANDLE value SMX_ACTOR

        Actor.Angle(SMX_ACTOR) = SMX_Kf


        if DYOS_VERSION > 300010
        then
            SMX_read_int() // B_ACTOR_FLAGS
            TEMP_FLAGS = SMX_A
            call @SMX_Array_Actor_Set 3 SMX_NUM_ACTORS B_ACTOR_FLAGS TEMP_FLAGS

            if lbit_is_true TEMP_FLAGS ACTOR_PROP_CROUCH
            then Task.PlayAnim(SMX_ACTOR,"WEAPON_crouch", "PED", 1000.0, true, false, false, false, -1)
            end

            SMX_read_int() // Weapon
            call @Weapon_GetModelID 1 SMX_A result SMX_B
            if SMX_B <> 0 
            then
                Model.Load(SMX_B)
                Model.Requested()

                repeat
                    Wait0()
                until Model.Available(SMX_B)

                Actor.setGiveWeaponAndAmmo(SMX_ACTOR, SMX_A, 999)
                
                Model.Release(SMX_B)
            end
        end

    end

case VM_TASK_KILL_PLAYER

    SMX_read_int() // id
    call @SMX_Get_Index_Actor 1 SMX_A result SMX_INDEX_ACTORS

    if SMX_INDEX_ACTORS <> 0
    then
        call @SMX_Array_Actor_Get 2 SMX_INDEX_ACTORS B_ACTOR_HANDLE value SMX_ACTOR
        
        if Actor.Defined(SMX_ACTOR)
        then
            Task.KillActorOnFoot(SMX_ACTOR, hPlayer)

            SMX_read_int()
            call @SMX_Array_Actor_Set 3 SPS_INDEX_ACTORS SMX_INDEX_ACTORS SMX_A
            Actor.SetWeaponAccuracy(SMX_ACTOR) = SMX_A

            SMX_read_float()
            call @SMX_Array_Actor_Set 3 SPS_INDEX_ACTORS SMX_INDEX_ACTORS SMX_Af
            Actor.SetMeleeAccuracy(SMX_ACTOR) = SMX_Af

            SMX_read_int()
            call @SMX_Array_Actor_Set 3 SPS_INDEX_ACTORS SMX_INDEX_ACTORS SMX_A
            Actor.SetAttackRate(SMX_ACTOR) = SMX_A
        end
    end



// 0x8000 - 0x8FFF : Command conditional

case VM_PLAYER_IS_POSITION
    SMX_read_3_float()
    // SMX_Af = X
    // SMX_Bf = Y
    // SMX_Cf = Z

    SMX_CONT_CONDITION += 1

    SMX_TEMP_OFFSET = SMX_CONT_CONDITION * 4

    if Actor.Defined(hPlayer)
    then

        2401: read_memory_with_offset SMX_BUFFER_TARGET_BLIPS offset SMX_TEMP_OFFSET size 4 store_to SMX_TARGET_BLIP
        2401: read_memory_with_offset SMX_BUFFER_TARGET_SPHERES offset SMX_TEMP_OFFSET size 4 store_to SMX_TARGET_SPHERE

        if and
            SMX_TARGET_BLIP == 0
            SMX_TARGET_SPHERE == 0
        then
            SMX_TARGET_BLIP = Blip.Create(SMX_Af, SMX_Bf, SMX_Cf, 1, 1)
            0165: set_marker SMX_TARGET_BLIP color_to 0
            0168: show_on_radar SMX_TARGET_BLIP size 3
            018B: show_on_radar SMX_TARGET_BLIP in 3

            SMX_TARGET_SPHERE = Sphere.Create(SMX_Af, SMX_Bf, SMX_Cf, 2.0)
            2402: write_memory_with_offset SMX_BUFFER_TARGET_BLIPS offset SMX_TEMP_OFFSET size 4 value SMX_TARGET_BLIP
            2402: write_memory_with_offset SMX_BUFFER_TARGET_SPHERES offset SMX_TEMP_OFFSET size 4 value SMX_TARGET_SPHERE

            SMX_NUM_TARGET_BLIPS += 1
            SMX_NUM_TARGET_SPHERES += 1
        end

        if Actor.IsLocated3D(hPlayer, 0, SMX_Af, SMX_Bf, SMX_Cf, 1.0, 1.0, 3.0)
        then
            SMX_IF_TRUE += 1

            Blip.Remove(SMX_TARGET_BLIP)
            Sphere.Remove(SMX_TARGET_SPHERE)
            2402: write_memory_with_offset SMX_BUFFER_TARGET_BLIPS offset SMX_TEMP_OFFSET size 4 value 0
            2402: write_memory_with_offset SMX_BUFFER_TARGET_SPHERES offset SMX_TEMP_OFFSET size 4 value 0
            SMX_NUM_TARGET_BLIPS -= 1
            SMX_NUM_TARGET_SPHERES -= 1
        end
    
        //printf "goto the red circle~n~ to %.2f %.2f %.2f" 84 SMX_Af SMX_Bf SMX_Cf
    else
        SMX_IF_TRUE += 1
        SMX_NUM_TARGET_BLIPS -= 1
        SMX_NUM_TARGET_SPHERES -= 1
    end


case VM_PLAYER_IS_KILL_TO_ACTOR

    SMX_read_int() // SMX_A = id
    call @SMX_Get_Index_Actor 1 id SMX_A index SMX_INDEX_ACTORS

    if SMX_INDEX_ACTORS <> 0
    then
        call @SMX_Array_Actor_Get 2 SMX_INDEX_ACTORS B_ACTOR_HANDLE value SMX_ACTOR

        SMX_CONT_CONDITION += 1
        SMX_TEMP_OFFSET = SMX_CONT_CONDITION
        SMX_TEMP_OFFSET *= 4
        
        if Actor.Defined(SMX_ACTOR)
        then
            SMX_ACTOR_TO_KILL += 1

            2401: read_memory_with_offset SMX_BUFFER_TARGET_BLIPS offset SMX_TEMP_OFFSET size 4 store_to SMX_TARGET_BLIP
            
            if SMX_TARGET_BLIP == 0
            then
                0187: SMX_TARGET_BLIP = create_marker_above_actor SMX_ACTOR
                0165: set_marker SMX_TARGET_BLIP color_to 0
                0168: show_on_radar SMX_TARGET_BLIP size 3
                018B: show_on_radar SMX_TARGET_BLIP in 3

                2402: write_memory_with_offset SMX_BUFFER_TARGET_BLIPS offset SMX_TEMP_OFFSET size 4 value SMX_TARGET_BLIP

                SMX_NUM_TARGET_BLIPS += 1
            end

            if Actor.IsDead(SMX_ACTOR)
            then
                SMX_IF_TRUE += 1
                SMX_ACTOR_TO_KILL -= 1

                Blip.Remove(SMX_TARGET_BLIP)
                2402: write_memory_with_offset SMX_BUFFER_TARGET_BLIPS offset SMX_TEMP_OFFSET size 4 value 0
                SMX_NUM_TARGET_BLIPS -= 1
            end
        else
            SMX_IF_TRUE += 1
            SMX_NUM_TARGET_BLIPS -= 1
        end
    end



    //   0x6000 : UI/UX

case VM_FADE
    SMX_read_int() // SMX_A = time
    10@ = SMX_A
    SMX_read_int() // SMX_A = in/out
    11@ = SMX_A

    fade 10@ 11@

case VM_PRINT_FORMATTED
    SMX_read_int() // SMX_A = time
    10@ = SMX_A
    SMX_read_byte() // SMX_A = lenght
    11@ = SMX_A

    if 11@ > 0
    then
        if 11@ > 256
        then 11@ = 256
        end
        
        call @label_addr 1 @buffer_STRING addr 7@
        
        SMX_TEMP_OFFSET = 0
        while SMX_TEMP_OFFSET < 11@

            SMX_read_byte() // SMX_A = Character
            write_memory_with_offset 7@ offset SMX_TEMP_OFFSET size 4 value SMX_A
            
            SMX_TEMP_OFFSET += 1
        end
        
        write_memory_with_offset 7@ offset SMX_TEMP_OFFSET size 4 value 0x00

        string_format 15@s = "%s" 7@
        printf "%s" 10@ 15@s
    end

default
    nop
    SMX_CURRENT_OFFSET -= 4
    printf "ERROR:~n~UNKNOW COMMAND: %X~n~OFFSET: %i" WAIT_ERROR FILE_COMMAND SMX_CURRENT_OFFSET
    call @waitPlis 1 WAIT_ERROR
    SMX_CURRENT_OFFSET = FILE_SIZE
end

return
