:Menu_Mission
NUMBER_PAGE = 1
CURRENT_PAGE = 1

Menu.SetColumn(hPanel, 0, 'M0000WR','M00000/','M00000[','M0000WT','M00007R',none,none,none,none,none,none,none,none)
//Menu.SetColumn(hPanel, 0, 'M0000WR','M00000/','M00000[','M0000WT','M00000{','M00007R',none,none,none,none,none,none,none)

if 0A9A: FILE_HANDLE = DIR_DYOS "ab"
then
	0A9C: FILE_SIZE = file FILE_HANDLE size

	0@ = FILE_SIZE
	if and
		SMX_IN_CONDITION_BLOCK == true
		SMX_IF_EMITTED == false
	then
		0@ += 4

		FILE_COMMAND = VM_IF
		SMX_write_command()
	end

	0A9B: file_close FILE_HANDLE
end

Menu.SetItemWithNumber(hPanel, 1, 0, 'M00000-', 0) // #0
Menu.SetItemWithNumber(hPanel, 1, 1, 'M00000]', 0@) // 4b

Menu_Take_Return()
Menu_Take_Selection()
repeat
	if Menu_Close_Forced()
	then jump @Menu_Close
	end

	if Menu_Enter()
	then
		setMenuClear()
		Menu.SetActiveItem(hPanel, 0)

		switch _SELECTED
			case 2 // Start
				if 0A9A: FILE_HANDLE = DIR_DYOS "rb"
				then
					Actor.GetPosition(hPlayer, SMX_Af, SMX_Bf, SMX_Cf)
					call @ClearArea_Radius 4 SMX_Af SMX_Bf SMX_Cf 100.0
					Actor.SetHealth(hPlayer) = 100

					FILE_OFFSET = 0
					if SMX_BUFFER_MAIN <> 0
					then 0AC9: free_allocated_memory SMX_BUFFER_MAIN
					end

					0A9C: FILE_SIZE = file FILE_HANDLE size

					0AC8: SMX_BUFFER_MAIN = allocate_memory_size FILE_SIZE

					while FILE_OFFSET < FILE_SIZE
						0AD5: file_seek FILE_HANDLE offset FILE_OFFSET origin 0
						0A9D: read_file FILE_HANDLE size 1 to 0@

						2402: write_memory_with_offset SMX_BUFFER_MAIN offset FILE_OFFSET size 1 value 0@
						FILE_OFFSET += 1
					end
					FILE_OFFSET = 0

					Fs.Close(FILE_HANDLE)
					Menu_Display_Hide()

					//printf "cd.. DYOS/#0" WAIT_DEFAULT
					//call @waitPlis 2 WAIT_DEFAULT false
					
					if FILE_SIZE == 0
					then
						printf "file not writed" WAIT_ERROR
						//call @waitPlis 2 WAIT_ERROR false
						goto @read_found_error
					else
						if or
							FILE_OFFSET == FILE_SIZE
							FILE_SIZE == 4
						then
							printf "file in white" WAIT_ERROR
							//call @waitPlis 2 WAIT_ERROR false
							goto @read_found_error
						end

						Display.SetRadar(true)
						if not Actor.IsDriving(hPlayer)
						then Actor.SetCollisionDetection(hPlayer) = true
						end
						Player.SetCanMove(hChar) = true
						
						// FILE_VERSION [XXX][XX]
						//
						// [XXX] : INT version
						// [XX] : INT build
						SMX_CURRENT_OFFSET = 0
						SMX_read_int()
						FILE_VERSION = SMX_A

						if bit_is_true DYOS_FLAGS DEBUG_MODE
						then
							printf "dyos_version: %i~n~file_version: %i" WAIT_INFO DYOS_VERSION FILE_VERSION
							call @waitPlis 1 WAIT_INFO

							int 0@ = 3000 // time
							int 1@ = 0 // in wait
							int 2@ // initial value
							int 3@ // last value
							
							2@ = Game.GetCurrentTimeMs()
						   
							while true
								3@ = Game.GetCurrentTimeMs()
								3@ -= 2@
								if 3@ > 0@
								then break
								end
								Wait0()
								3@ /= 1000
								1@ = 3 - 3@
								printf "Start mission on... %i" 10 1@
							end
						else
							fade 0 WAIT_INFO
							call @waitPlis 1 WAIT_INFO
							fade 1 0
						end

						SMX_IF_ACTIVE = 0
						SMX_IF_TRUE = 0
						SMX_CONT_CONDITION = 0
						SMX_MISSION_FAIL = false

						SMX_NUM_TARGET_BLIPS = 0
						SMX_NUM_TARGET_SPHERES = 0
						SMX_NUM_ACTORS = 0
						SMX_INDEX_ACTORS = 0
						
						if SMX_BUFFER_ACTORS <> 0
						then
							0AC9: free_allocated_memory SMX_BUFFER_TARGET_BLIPS
							0AC9: free_allocated_memory SMX_BUFFER_TARGET_SPHERES
							0AC9: free_allocated_memory SMX_BUFFER_ACTORS
						end
						
						0@ = MAX_CONDITIONS * 4
						0AC8: SMX_BUFFER_TARGET_BLIPS = allocate_memory_size 0@
						call @MemoryWrite 3 SMX_BUFFER_TARGET_BLIPS 0@ 0
						0AC8: SMX_BUFFER_TARGET_SPHERES = allocate_memory_size 0@
						call @MemoryWrite 3 SMX_BUFFER_TARGET_SPHERES 0@ 0

						0@ = MAX_ITEMS_BY_TYPE * 4 // handles
						0@ = 0@ * ACTOR_NUM_PROPERTIES // for save props
						0AC8: SMX_BUFFER_ACTORS = allocate_memory_size 0@
						call @MemoryWrite 3 SMX_BUFFER_ACTORS 0@ 0

						SMX_ACTOR_TO_KILL = 0

						hPlayer = Player.GetActor(0)
						hGroup = Player.GetGroup(0)

						while SMX_CURRENT_OFFSET < FILE_SIZE
							wait0()

							// Leer comando
							SMX_read_cmd()

							SMX_read_command()
							if SMX_MISSION_FAIL == true
							then 
								while Actor.IsDead(hPlayer)
									wait0()
								end
								fade 1 0
								break
							end
						end

						call @DebugLog 2 "smx_clear: buffer target spheres" WAIT_DEBUG
						call @clear_buffer 1 BUFFER_TARGET_SPHERES

						call @DebugLog 2 "smx_clear: buffer targets blips" WAIT_DEBUG
						call @clear_buffer 1 BUFFER_TARGET_BLIPS

						call @DebugLog 2 "smx_clear: buffer actor" WAIT_DEBUG
						call @clear_buffer 1 BUFFER_ACTORS

						call @DebugLog 2 "smx_clear: reset map" WAIT_DEBUG
						Text.ClearHelp()
						Display.SetRadar(false)
						if not Actor.IsDriving(hPlayer)
						then Actor.SetCollisionDetection(hPlayer) = false
						end
						Player.SetCanMove(hChar) = false
						
						Player_RemindPosition()
						call @ClearArea_Radius 4 Player_Memoize_pos_x Player_Memoize_pos_y Player_Memoize_pos_z 100.0
					end

					printf "finish, closed file" WAIT_INFO
					call @waitPlis 1 WAIT_INFO
					
					Menu_Display_Show()
					jump @Menu_Mission
				else
					printf "file not exist" WAIT_ERROR
					call @waitPlis 2 WAIT_ERROR false
				end

				Menu.SetActiveItem(hPanel, 2)
				jump @Menu_Mission

			case 3 // Reset

				Menu.SetColumn(hPanel, 0, 'M0000A1','M0000A0','M00009Z',none,none,none,none,none,none,none,none,none,none)
				Menu_Take_Return()
				Menu_Take_Selection()
				repeat
					if Menu_Enter()
					then
						if _SELECTED == 0
						then goto @Menu_Mission
						end

						if _SELECTED == 1
						then
							reset_file()
							Menu.SetItemWithNumber(hPanel, 1, 1, 'M00000]', 4) // 4b
							break
						end
					end
				until Menu_Return()

				Menu.SetActiveItem(hPanel, 4)
				jump @Menu_Mission
			default
				Text.PrintHelp('M0000WQ') // fuera de rango
				Menu.SetActiveItem(hPanel, _SELECTED)
				jump @Menu_Mission
		end//switch _SELECTED
	end
until Menu_Return()  
setMenuClear()
Menu.SetActiveItem(hPanel, MENU_MISSION)
jump @Menu_iMain
