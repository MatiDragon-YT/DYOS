:File_Chech_Exist
    if CURRENT_PLATFORM == __PLATFORM_WINDOWS
    then DIR_DYOS = "CLEO/0.smx"
    else DIR_DYOS = "/0.smx"
    end
    if 0AAB:  does_file_exist DIR_DYOS
    then
        FILE_EXIST = true
        
        0A9A: FILE_HANDLE = DIR_DYOS "rb"
        0A9C: FILE_SIZE = file FILE_HANDLE size
        0A9B: file_close FILE_HANDLE
    else
        if 0A9A: FILE_HANDLE = DIR_DYOS "rb"
        then
            Fs.Close(FILE_HANDLE)
            FILE_EXIST = true
        else
            Text.PrintHelp('M00001?') // ~p~SMX:~s~~n~~t~Creando archivo "./0.smx".

            if 0A9A: FILE_HANDLE = DIR_DYOS "w+b"
            then
                call @waitPlis 1 150
                0A9E: write_file FILE_HANDLE size 4 from 0 // 300010
                call @waitPlis 1 150
                Fs.Close(FILE_HANDLE)
                call @waitPlis 1 150

                FILE_EXIST = true

                if 0AAB:  does_file_exist DIR_DYOS
                then
                    reset_file()
                    Text.ClearHelp()
                    Text.PrintHelp('M00001_') // ~p~SMX:~s~~n~~t~Archivo creado.
                    goto @File_Chech_Exist
                end
            end

            FILE_EXIST = false
            Text.PrintHelp('M00000+') // ~p~SMX:~s~~n~~t~Error con "./0.smx".
        end
    end
return

:clear_buffer
    // 0@ buffer
    // 1@ size
    // 2@ type

    int 0@
    int 1@

    int 5@ = 0
    int 6@ = 0
    
    fixMagicPointer()
    if 0@ == BUFFER_ACTORS
    then
        while SMX_NUM_ACTORS > 0
            call @SMX_Array_Actor_Get 2 SMX_NUM_ACTORS B_ACTOR_HANDLE result SMX_ACTOR

            if Actor.Defined(SMX_ACTOR)
            then Actor.Remove(SMX_ACTOR)
            end

            call @SMX_Array_Actor_Set 3 SMX_NUM_ACTORS B_ACTOR_HANDLE value 0

            SMX_NUM_ACTORS -= 1
        end
        goto @exit_clear_buffer
    end

    if 0@ == BUFFER_TARGET_BLIPS
    then
        1@ = 0
        while 1@ < MAX_CONDITIONS
            SMX_TEMP_OFFSET = 1@ * 4
            2401: read_memory_with_offset SMX_BUFFER_TARGET_BLIPS offset SMX_TEMP_OFFSET size 4 store_to SMX_TARGET_BLIP

            if Blip.Defined(SMX_TARGET_BLIP)
            then Blip.Remove(SMX_TARGET_BLIP)
            end

            2402: write_memory_with_offset SMX_BUFFER_TARGET_BLIPS offset SMX_TEMP_OFFSET size 4 value 0

            1@ += 1
        end
        goto @exit_clear_buffer
    end

    if 0@ == BUFFER_TARGET_SPHERES
    then
        1@ = 0
        while 1@ < MAX_CONDITIONS
            SMX_TEMP_OFFSET = 1@ * 4
            2401: read_memory_with_offset SMX_BUFFER_TARGET_SPHERES offset SMX_TEMP_OFFSET size 4 store_to SMX_TARGET_SPHERE

            if SMX_TARGET_SPHERE <> 0
            then Sphere.Remove(SMX_TARGET_SPHERE)
            end

            2402: write_memory_with_offset SMX_BUFFER_TARGET_SPHERES offset SMX_TEMP_OFFSET size 4 value 0

            1@ += 1
        end
        goto @exit_clear_buffer
    end

    :exit_clear_buffer
