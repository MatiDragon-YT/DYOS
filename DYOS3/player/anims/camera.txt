Menu.SetActiveItem(hPanel, 0)
NUMBER_PAGE = 2
CURRENT_PAGE = 1
:PlayerAnimsCamera
FrenarPaginacion()

Text.PrintHelpForever('M0000AW')
Animation.Load('CAMERA')
while not Animation.Available('CAMERA')
	WaitLoadingAtMenu()
end
Text.ClearHelp()

if
	CURRENT_PAGE == 1
then
	Menu.SetColumn(hPanel,0,'M0000DL', 'M0000JU', 'M0000JV', 'M0000JW', 'M0000JX', 'M0000JY', 'M0000JZ', 'M0000K0', 'M0000K1', none, none, none, none)
	Menu_Take_Selection()
	repeat
		
		if VK_Right()
		then jump @PlayerAnimsCamera
		end
		if Menu_Enter()
		then
			ResetAnimPlayer()

			IFP_FILE = "CAMERA"
			if _SELECTED == 0
			then IFP_ANIM = "CAMCRCH_CMON"
			end
			if _SELECTED == 1
			then IFP_ANIM = "CAMCRCH_IDLELOOP"
			end
			if _SELECTED == 2
			then IFP_ANIM = "CAMCRCH_STAY"
			end
			if _SELECTED == 3
			then IFP_ANIM = "CAMCRCH_TO_CAMSTND"
			end
			if _SELECTED == 4
			then IFP_ANIM = "CAMSTND_CMON"
			end
			if _SELECTED == 5
			then IFP_ANIM = "CAMSTND_IDLELOOP"
			end
			if _SELECTED == 6
			then IFP_ANIM = "CAMSTND_LKABT"
			end
			if _SELECTED == 7
			then IFP_ANIM = "CAMSTND_TO_CAMCRCH"
			end
			_SELECTED_BEFORE = _SELECTED
			Player_IFP_Set()
			Menu_Take_Selection()
		end
	until Menu_Return()
end

if
	CURRENT_PAGE == 2
then
	Menu.SetColumn(hPanel, 0, 'M0000DL', 'M0000K2', 'M0000K3', 'M0000K4', 'M0000K5', 'M0000K6', 'M0000K7', none, none, none, none, none, none)
	Menu_Take_Selection()
	repeat
		   
		if VK_Left()
		then jump @PlayerAnimsCamera
		end
		if Menu_Enter()
		then
			ResetAnimPlayer()
			
			IFP_FILE = "CAMERA"
			if _SELECTED == 0
			then IFP_ANIM = "PICCRCH_IN"
			end
			if _SELECTED == 1
			then IFP_ANIM = "PICCRCH_OUT"
			end
			if _SELECTED == 2
			then IFP_ANIM = "PICCRCH_TAKE"
			end
			if _SELECTED == 3
			then IFP_ANIM = "PICSTND_IN"
			end
			if _SELECTED == 4
			then IFP_ANIM = "PICSTND_OUT"
			end
			if _SELECTED == 5
			then IFP_ANIM = "PICSTND_TAKE"
			end

			_SELECTED_BEFORE = _SELECTED
			Player_IFP_Set()
			Menu_Take_Selection()
		end
	until Menu_Return()
end
Animation.Remove('CAMERA')
