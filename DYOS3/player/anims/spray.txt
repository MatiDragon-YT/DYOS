CURRENT_PAGE = 1
NUMBER_PAGE = 1
Menu.SetActiveItem(hPanel, 0)
Menu.SetColumn(hPanel, 0, 'M00002L', 'M00002M', 'M00002N', none, none, none, none, none, none, none, none, none, none)
Menu_Take_Selection()

Text.PrintHelpForever('M0000AW')
Animation.Load("SPRAYCAN")
repeat
	Wait0()
until Animation.Available("SPRAYCAN")
Text.ClearHelp()

repeat
	
	if Menu_Enter()
	then
		ResetAnimPlayer()

		if bit_is_true PLAYER_FLAGS PLAYER_PROP_COLISION
		then 
			if _SELECTED == 0
			then Task.PlayAnimSecondary(hPlayer, "SPRAYCAN_FIRE", "SPRAYCAN", 1000.0, true, false, false, false, -1)
			end
			if _SELECTED == 1
			then Task.PlayAnimSecondary(hPlayer, "SPRAYCAN_FULL", "SPRAYCAN", 1000.0, true, false, false, false, -1)
			end
		else
			if _SELECTED == 0
			then Task.PlayAnim(hPlayer, "SPRAYCAN_FIRE", "SPRAYCAN", 1000.0, true, true, true, false, -1)
			end
			if _SELECTED == 1
			then Task.PlayAnim(hPlayer, "SPRAYCAN_FULL", "SPRAYCAN", 1000.0, true, true, true, false, -1)
			end
		end
		Menu_Take_Selection()
	end
until Menu_Return()
Animation.Remove("SPRAYCAN")
