if CURRENT_PAGE == 1
then
	Menu.SetColumn(hPanel,0,'M0000IQ','M0000HI','M0000HJ','M0000HK','M0000HL','M0000HM','M0000HN','M0000HO','M0000HP','M0000HQ','M0000HR','M0000HS','M0000HT')
	Menu_Take_Return()
	Menu_Take_Selection() 
	repeat
		wait 0
	    if Menu_Close_Forced()
	    then jump @Menu_Close
	    end
		if VK_Right()
		then jump @PlayerModel
		end
		if Menu_Enter()
		then
			int 15@ = 0
			15@ += _SELECTED
			LoadSpecialActor()
			Text.PrintHelpForever('M0000AW')
			repeat
				wait 0
			until SpecialActor.Available(1)
			Text.ClearHelp()
			Player.SetModel(hChar, #SPECIAL01)
			SpecialActor.Release(1)

			15@ *= -1
			call @bset_var 2 B_VAR_PLAYER_MODEL = 15@

			Menu_Take_Selection()
		end
		if ParaEspecial()
		then
			if $numActores < 50
			then
				$indActor = $numActores
				$numActores++
				$indActor++
				
				6@ = Actor.GetModel(hPlayer)
				GenerarActor()
				Aplicated()
				Menu.SetActiveItem(hPanel, 0)
				jump @MenuDeEdicionDelActor
			end
		end
	until Menu_Return() 
end
