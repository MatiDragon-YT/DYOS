                then
                    FILE_OFFSET = 0
                    if SMX_BUFFER_MAIN <> 0
                    then 0AC9: free_allocated_memory SMX_BUFFER_MAIN
                    end

                    0A9C: FILE_SIZE = file FILE_HANDLE size

                    0AC8: SMX_BUFFER_MAIN = allocate_memory_size FILE_SIZE

                    while FILE_OFFSET < FILE_SIZE
                        0AD5: file_seek FILE_HANDLE offset FILE_OFFSET origin 0
                        0A9D: read_file FILE_HANDLE size 1 to 0@

                        2402: write_memory_with_offset SMX_BUFFER_MAIN offset FILE_OFFSET size 1 value 0@
                        FILE_OFFSET += 1
                    end
                    FILE_OFFSET = 0

                    Fs.Close(FILE_HANDLE)
                    Menu_Display_Hide()

                    //printf "cd.. DYOS/#0" WAIT_DEFAULT
                    //call @waitPlis 2 WAIT_DEFAULT false
                    
                    if FILE_SIZE == 0
                    then
                        printf "file not writed" WAIT_ERROR
                        //call @waitPlis 2 WAIT_ERROR false
                        goto @read_found_error
                    else
                        if or
                            FILE_OFFSET == FILE_SIZE
                            FILE_SIZE == 4
                        then
                            printf "file in white" WAIT_ERROR
                            //call @waitPlis 2 WAIT_ERROR false
                            goto @read_found_error

                            printf "Start mission on... %i" 10 1@

                        //

                        // [XX] : INT build

                        FILE_VERSION = SMX_A

                        SMX_IF_ACTIVE = 0
                        SMX_ENTER_CONDITION = false
                        SMX_COMMAND_IS_CONDITIONAL = 0
                        SMX_COND_START_OFFSET  = 0

                        SMX_NUM_ACTORS = 0
                        SMX_INDEX_ACTORS = 0

                        SMX_ACTOR_TO_KILL = 0

                        hGroup = Player.GetGroup(0)

                            wait0()

                            printf "leeyendo" 250
                            call @waitPlis 1 250

                            read_buffer_cmd()

                            SMX_read_command()
                            printf "aplicado" 250
                            call @waitPlis 1 250
                        end

                        printf "Limpiando memoria" 2222
                        call @waitPlis 1 2222

                        call @clear_buffer 3 SMX_BUFFER_BLIPS MAX_CONDITIONS BUFFER_BLIPS

                    jump @Menu_Mission

:File_Chech_Exist
    if CURRENT_PLATFORM == __PLATFORM_WINDOWS
    then DIR_DYOS = "CLEO/DYOS/#0"
    else DIR_DYOS = "/DYOS/#0"
    end
    if 0AAB:  does_file_exist DIR_DYOS
    then
        FILE_EXIST = true
        
        0A9A: FILE_HANDLE = DIR_DYOS "rb"
        0A9C: FILE_SIZE = file FILE_HANDLE size
        0A9B: file_close FILE_HANDLE
    else
        if 0A9A: FILE_HANDLE = DIR_DYOS "rb"
        then
            Fs.Close(FILE_HANDLE)
            FILE_EXIST = true
        else
            Text.PrintHelp('M00001?') // ~p~SMX:~s~~n~~t~Creando archivo "./DYOS/#0".

            if CURRENT_PLATFORM == __PLATFORM_WINDOWS
            then
                if 8AE4: directory_exist "CLEO/DYOS/"
                then
                    if 0AE5: create_directory "CLEO/DYOS/" //IF and SET
                    then call @waitPlis 1 150
                    end
                end
            else
                if 8AE4: directory_exist "DYOS/"
                then
                    if 0AE5: create_directory "DYOS/" //IF and SET
                    then call @waitPlis 1 150
                    end
                end
            end

            if 0A9A: FILE_HANDLE = DIR_DYOS "w+b"
            then
                call @waitPlis 1 150
                0A9E: write_file FILE_HANDLE size 4 from 0 // 300010
                call @waitPlis 1 150
                Fs.Close(FILE_HANDLE)
                call @waitPlis 1 150

                FILE_EXIST = true

                if 0AAB:  does_file_exist DIR_DYOS
                then
                    reset_file()
                    Text.ClearHelp()
                    Text.PrintHelp('M00001_') // ~p~SMX:~s~~n~~t~Archivo creado.
                    goto @File_Chech_Exist
                end
            end

            FILE_EXIST = false
            Text.PrintHelp('M00000+') // ~p~SMX:~s~~n~~t~Error con "./DYOS/#0".
        end
    end
return

:read_buffer_cmd
    2401: read_memory_with_offset SMX_BUFFER_MAIN offset SMX_CURRENT_OFFSET size 4 store_to FILE_COMMAND
    SMX_CURRENT_OFFSET += 4

    printf "cmd: %i" 155 FILE_COMMAND
    call @waitPlis 1 155
return

:read_buffer_Value
    2401: read_memory_with_offset SMX_BUFFER_MAIN offset SMX_CURRENT_OFFSET size 4 store_to SMX_A
    SMX_CURRENT_OFFSET += 4

    printf "val: %i" 155 SMX_A
    call @waitPlis 1 155
return

:read_buffer_Vec3
    2401: read_memory_with_offset SMX_BUFFER_MAIN offset SMX_CURRENT_OFFSET size 4 store_to SMX_A
    SMX_CURRENT_OFFSET += 4
    2401: read_memory_with_offset SMX_BUFFER_MAIN offset SMX_CURRENT_OFFSET size 4 store_to SMX_B
    SMX_CURRENT_OFFSET += 4
    2401: read_memory_with_offset SMX_BUFFER_MAIN offset SMX_CURRENT_OFFSET size 4 store_to SMX_C
    SMX_CURRENT_OFFSET += 4

    printf "vec3: %i %i %i" 155 SMX_A SMX_B SMX_C
    call @waitPlis 1 155
return

:read_found_error

:reset_file
    if 0A9A: FILE_HANDLE = DIR_DYOS "wb"
    then
        FILE_OFFSET = 0
        
        // FILE_VERSION [XXX][XX]
        //
        // [XXX] : INT version
        // [XX] : INT build
        0A9E: write_file FILE_HANDLE size 4 from DYOS_CODE_VERSION
        Fs.Close(FILE_HANDLE)
    end
return

:clear_buffer
    // 0@ buffer
    // 1@ size
    // 2@ type

    int 0@
    int 1@

    int 5@ = 0
    int 6@ = 0
    while true
        if 0@ == BUFFER_ACTORS
        then
            fixMagicPointer()
            while SMX_NUM_ACTORS > 0
                
                SMX_TEMP_OFFSET = SMX_NUM_ACTORS * ACTOR_NUM_PROPERTIES
                SMX_TEMP_OFFSET *= 4
                2401: read_memory_with_offset SMX_BUFFER_ACTORS offset SMX_TEMP_OFFSET size 4 store_to SMX_ACTOR
                if Actor.Defined(SMX_ACTOR)
                then Actor.Remove(SMX_ACTOR)
                end
                2402: write_memory_with_offset SMX_BUFFER_ACTORS offset SMX_TEMP_OFFSET size 4 value 0

                SMX_NUM_ACTORS -= 1
            end
            goto @exit_clear_buffer
        end
        if 0@ == BUFFER_ACTORS_DEADS
        then
            fixMagicPointer()
            while SMX_NUM_ACTORS > 0
                
                SMX_TEMP_OFFSET = SMX_NUM_ACTORS * ACTOR_NUM_PROPERTIES
                SMX_TEMP_OFFSET *= 4
                2401: read_memory_with_offset SMX_BUFFER_ACTORS offset SMX_TEMP_OFFSET size 4 store_to SMX_ACTOR
                if Actor.IsDead(SMX_ACTOR)
                then
                    Actor.Remove(SMX_ACTOR)
                    2402: write_memory_with_offset SMX_BUFFER_ACTORS offset SMX_TEMP_OFFSET size 4 value 0
                end

                SMX_NUM_ACTORS -= 1
            end
            goto @exit_clear_buffer
        end

        while 5@ < 1@
            6@ = 5@ * 4

            2401: read_memory_with_offset 0@ offset 6@ size 4 store_to 10@
            
            //if 2@ == BUFFER_MARKERS
            //then Blip.Remove(10@)
            //end
            if 2@ == BUFFER_SPHERES
            then Sphere.Remove(10@)
            end
            if 2@ == BUFFER_BLIPS
            then Blip.Remove(10@)
            end
            if 2@ == BUFFER_VEHICLES
            then Car.Remove(10@)
            end
            if 2@ == BUFFER_OBJECTS
            then Object.Remove(10@)
            end
            if 2@ == BUFFER_PARTICLES
            then Particle.Remove(10@)
            end
            if 2@ == BUFFER_PICKUPS
            then Pickup.Remove(10@)
            end
            if 2@ == BUFFER_SEARCHLIGHT
            then Searchlight.Remove(10@)
            end


            2402: write_memory_with_offset 0@ offset 6@ size 4 value 0
            5@ += 1
        end
        goto @exit_clear_buffer
    end
    :exit_clear_buffer
ret 0
