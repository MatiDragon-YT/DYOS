:Buffer_Var_Set
    int 0@,1@,2@
// 0@ INT[1-255] : index
// 1@ INT : value
	0@ *= 4

	call @label_addr 1 @temp_var 2@

    2@ += 0@

    0DD5: CURRENT_PLATFORM = get_platform
	if CURRENT_PLATFORM == __PLATFORM_WINDOWS
	then Memory.Write(2@, 4, 1@, false)
	else 0DD9: write_mem_addr 2@ value 1@ size 4 add_ib 0 protect 0
	end
ret 0
