:CrearActor
	FrenarSeleccion()
	int 0@ = B_VAR_MODEL_ACTOR
	bget_var()
	int 6@ = 0@

	$indActor = $numActores
	if $numActores < 50
	then
		int 0@ = 1
		while 0@ <= $numActores
			$indActor = 0@
			if Actor.Defined($Actor($indActor,255i))
			then
				0@ += 1
			else
				0@ *= -1
				break
			end
		end
		if 0@ < 0
		then
			0@ *= -1
		else
			$indActor = $numActores
			$numActores++
			$indActor++
		end
		CargarActorModelo()
		EsconderMenu()
		30@ = false
		25@ = 2
		19@ = 0
		repeat
			wait 0
			iluminar_al_jugador()
			FixUpdate()
			
			if ParaEspecial()
			then
				if 19@ == 0
				then 19@ = 1
				else 19@ = 0
				end
				FrenarEspecial()
			end

			if 19@ == 1
			then InputNumberAndroid()
			else
				if $ModoDeCompilacion == f_PC
				then InputNumberPC()
				end
				Text.PrintWithNumberNow('M000007', 6@, 200, 0)

				if 30@ == false
				then
					if $ModoDeCompilacion == f_Android
					then Text.PrintHelp('M0000GB')
					else Text.PrintHelp('M0000GC')
					end
					30@ = true
				end
				int 12@ = 6@
				6@ = 1
				repeat
					6@+=1
					ModeloSiguiente_Actor()
					Model.Release(6@)
				until 6@ == 0
				6@ = 12@
				if ParaIzquierda()
				then
					Model.Release(6@)
					6@--
					ModeloAnterior_Actor()
					CargarActorModelo()
				end
				if ParaDerecha()
				then
					Model.Release(6@)
					6@++
					ModeloSiguiente_Actor()
					CargarActorModelo()
				end
				if ParaAtras()
				then
					Model.Release(6@)
					6@ -= 10
					ModeloAnterior_Actor()
					CargarActorModelo()
				end
				if ParaAdelante()
				then
					Model.Release(6@)
					6@ += 10
					ModeloSiguiente_Actor()
					CargarActorModelo()
				end
			end

			param1r = false
			if ParaDecrecer()
			then
				if 6@ >= 0
				then
					6@ = Actor.Model(_Player)
					GenerarActor()
					recuperarSkinDelPlayer()
					param1r = true
					break
				end
			end
			
			if Selecionas()
			then
				param1r = true
				if 6@ >= 0
				then
					if not Actor.Model(_Player) == 6@
					then
						CargarActorModelo()
					end
					GenerarActor()
				else
					int 15@ = 6@
					15@ *= -1

					call @CustomSkin4 1 15@
					repeat
						wait 0
						FixUpdate()
					until SpecialActor.Available(1)
					GetUpdatePlayerPosition()
					$Actor($indActor,255i) = Actor.Create(PedType.Special, #SPECIAL01, $tempvar_X_coord, $tempvar_Y_coord, $tempvar_Z_coord)
					Actor.Angle($Actor($indActor,255i)) = $tempvar_Angle
					Actor.SetLinkInInterior($Actor($indActor,255i), $Active_Interior)
					SpecialActor.Release(1)
					Actor_ResetProperties()

					int 0@ = B_ACTOR_TYPE
					int 1@ = $indActor
					int 2@ = 23
					bset_actor()
					
					int 0@ = B_ACTOR_GENERATION
					int 1@ = $indActor
					int 2@ = GENERATION_CUSTOM
					bset_actor()
					int 0@ = B_ACTOR_MODEL_SPECIAL
					int 1@ = $indActor
					int 2@ = 15@
					bset_actor()
				end
				break
			end
			if ParaAbajo()
			then
				param1r = undefined
				break
			end
		until SalirDelMenu()

		AparecerMenu()
		Menu.SetActiveItem($PANEL, 0)
		Text.ClearHelp()
		Text.ClearPrints()
	else
		Text.PrintHelp('M00000M')   
	end
	recuperarSkinDelPlayer()
return

:CargarActorModelo
	Model.Load(6@)
	Text.PrintHelpForever('M0000AW')
	if $ModoDeCompilacion == f_PC
	then
		TIMERB = 0
		while TIMERB <= 84
			Wait0()
			iluminar_al_jugador()
		end
	else
		TIMERZ = 0
		while TIMERZ <= 84
			Wait0()
			iluminar_al_jugador()
		end
	end
	repeat
		Wait0()
		iluminar_al_jugador()
	until Model.Available(6@)
	Text.ClearHelp()
	Player.SetModel(_Char,6@)
return

	Actor.Angle($Actor($indActor,255i)) = $tempvar_Angle
	Actor.SetLinkInInterior($Actor($indActor,255i), $Active_Interior)
	Actor_ResetProperties()

	Text.PrintWithNumberNow('M000007', 6@, 1000, 0)
return
