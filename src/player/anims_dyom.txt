int $LAST_ID = 344

EsconderMenu()
FrenarSeleccion()
repeat
	Wait0()

	if ParaIzquierda()
	then
		Text.PrintHelpForever('M0000AW')
		$CURRENT_ANIM_ID -= 1
		if $CURRENT_ANIM_ID < 0
		then $CURRENT_ANIM_ID = $LAST_ID
		end
		FrenarIzquierda()
	end
	if ParaAtras()
	then
		Text.PrintHelpForever('M0000AW')
		$CURRENT_ANIM_ID -= 10
		if $CURRENT_ANIM_ID < 0
		then $CURRENT_ANIM_ID = $LAST_ID
		end
		FrenarAtras()
	end
	if ParaDerecha()
	then
		Text.PrintHelpForever('M0000AW')
		$CURRENT_ANIM_ID += 1
		if $CURRENT_ANIM_ID > $LAST_ID
		then $CURRENT_ANIM_ID = 0
		end
		FrenarDerecha()
	end
	if ParaAdelante()
	then
		Text.PrintHelpForever('M0000AW')
		$CURRENT_ANIM_ID += 10
		if $CURRENT_ANIM_ID > $LAST_ID
		then $CURRENT_ANIM_ID = 0
		end
		FrenarAdelante()
	end

	AnimDyom()

	if $ANIM_NOT_IS_OF_PED == false
	then
		if $colPlayer == true
        then Task.PlayAnimSecondary(_Player, v$IFP_ANIM, "PED", 4.0, true, false, false, false, -1)
        else Task.PlayAnim(_Player, v$IFP_ANIM, "PED", 4.0, true, true, true, false, -1)
        end
		Text.ClearHelp()
	else
		Animation.Load(v$IFP_FILE)
		repeat
			Wait0()
		until Animation.Available(v$IFP_FILE)
		Text.ClearHelp()
		if $colPlayer == true
        then Task.PlayAnimSecondary(_Player, v$IFP_ANIM, v$IFP_FILE, 4.0, true, false, false, false, -1)
        else Task.PlayAnim(_Player, v$IFP_ANIM, v$IFP_FILE, 4.0, true, true, true, false, -1)
        end
		Animation.Remove(v$IFP_FILE)
		Game.PrintString(v$IFP_ANIM)
	end

	
	if $ModoDeCompilacion == ePara_PC
	then printf "%i~n~%s/%s" 100 $CURRENT_ANIM_ID v$IFP_FILE v$IFP_ANIM
	else Text.PrintWithNumberNow('M000007', $CURRENT_ANIM_ID, 100, 0)
	end

	if Selecionas()
	then break
	end
until SalirDelMenu()
AparecerMenu()
