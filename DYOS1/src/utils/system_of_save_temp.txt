:bset_actor
// 0@ INT : property
// 1@ INT[1-255] : actor
// 2@ INT : value

	int 0@ *= 4

	int 1@ *= 13
	int 1@ *= 4
	
	0@ += 1@ // result

	if $ModoDeCompilacion == ePara_PC
	then 0AC6: 3@ = label_addr @buf_actors
	else 0DD0: 3@ = label_addr @buf_actors
	end

    int 3@ += 0@

	if $ModoDeCompilacion == ePara_PC
	then Memory.Write(3@, 4, 2@, false)
	else 0DD9: write_mem_addr 3@ value 2@ size 4 add_ib 0 protect 0
	end
return
