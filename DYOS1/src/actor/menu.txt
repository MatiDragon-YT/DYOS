Menu.SetActiveItem($PANEL, 0)
Text.PrintHelp('M000009') // El modelo que elijas, se guarda para la proxima creacion por si necesitas duplicarlo.
:MenuDelActor
if
	$numActores >= 1
then
	if $indActor == 0
	then $indActor = 1
		if not Actor.Defined($Actor($indActor,255i))
		then ActorDefinidoSuperior()
		end
	end
	if Actor.Defined($Actor($indActor,255i))
	then
		10@s = 'M0000M9' // Editar

		if
			$numActores >= 2
		then
			15@s = 'M0000RV' // Editar todos
			20@s = 'M00000D' // Selecionar
		else
			15@s = 'DUMMY'
			20@s = 'DUMMY'
		end
	else
		$numActores--
		Actor.Remove($Actor($indActor,255i))
		$indActor--
		Wait0()
		jump @MenuDelActor
	end
else
	10@s = 'DUMMY'
	15@s = 'DUMMY'
	20@s = 'DUMMY'
	Menu.SetActiveItem($PANEL, 0)
end

Menu.SetColumn($PANEL,0,'M0000DO','M0000RX',10@s,15@s,20@s,none,none,none,none,none,none,none,none)
FrenarRetorno()
FrenarSeleccion()
repeat
	wait 0
	if EntrarAlMenu()
	then
		if _Selected == 0 // Create
		then
			Menu.SetActiveItem($PANEL, 0)
			:MenuQueActorCrear
			Menu.SetColumn($PANEL,0,'M0000DO','M0000AH','M0000IQ','M0000MU',none,none,none,none,none,none,none,none,none)
			FrenarRetorno()
			FrenarSeleccion()
			repeat
				Wait0()
				if EntrarAlMenu()
				then
					if _Selected == 0 // Ped
					then
						CrearActor()
						if param1r == true
						then jump @MenuDeEdicionDelActor
						end
						if param1r == false
						then jump @ActorDestroy
						end
						if param1r == undefined
						then jump @MenuQueActorCrear
						end
					end
					if _Selected == 1 // Especial
					then
						Menu.SetActiveItem($PANEL, 0)
						jump @ParaCrearActorEspecial
					end
					if _Selected == 2 // Custom
					then
						Menu.SetActiveItem($PANEL, 0)
						jump @ParaCrearActorPersonal
					end
				end
			until SalirDelMenu()
			Menu.SetActiveItem($PANEL, 0)
			jump @MenuDelActor
		end

		if _Selected == 1 // Edit
		then
			{$I ./edit.txt}
			Menu.SetActiveItem($PANEL, 1)
			jump @MenuDelActor
		end

		if _Selected == 2 // Edit all
		then
			{$I ./edit_all.txt}
			Menu.SetActiveItem($PANEL, 2)
			jump @MenuDelActor
		end

		if _Selected == 3 // Select
		then
			if $ModoDeCompilacion == ePara_PC
			then Text.PrintHelp('M0000JK')
			else Text.PrintHelp('M0000JL')
			end
			EsconderMenu()
			FrenarSeleccion()
			repeat
				Wait0()
				Actor_GetDataAtDisplay()
				Text.PrintWith2NumbersNow('M0000BG',$indActor,$numActores,150,0)

				if ParaEspecial()
				then
					$indActor = 0
					float 10@ = 1.0
					int 12@ = false
					while true 
						Wait0()
						Text.Print('M0000MX',10,0)
						12@ = true
						$indActor++
						if $indActor > $numActores
						then
							$indActor = 1
							10@ += 1.0
							if 10@ > 75.0
							then
								Text.ClearHelp()
								Text.PrintHelp('M0000L6')
								12@ = false
								break
							end
						end
						if Actor.Defined($Actor($indActor,255i))
						then
							if Actor.IsLocatedNearActor3D(_Player,0,$Actor($indActor,255i),10@,10@,10@)
							then break
							end
						end
					end
					Blip.Remove($ResaltadorDeEntidad)
					if 12@ == true
					then
						AparecerMenu()
						Menu.SetActiveItem($PANEL, 0)
						Text.ClearHelp()
						Text.ClearPrints()
						jump @MenuDeEdicionDelActor
					end
				end

				if Blip.Defined($ResaltadorDeEntidad)
				then
					Wait0()
					Actor_GetDataAtDisplay()
				else
					$ResaltadorDeEntidad = Blip.CreateAboveActor($Actor($indActor,255i))
					while ParaIzquierda()
						Wait0()
						Actor_GetDataAtDisplay()
						FixUpdate()
					end
					while ParaDerecha()
						Wait0()
						Actor_GetDataAtDisplay()
						FixUpdate()
					end
				end

				if ParaIzquierda()
				then
					ActorDefinidoInferior()
					Blip.Remove($ResaltadorDeEntidad)
				end
				if ParaDerecha()
				then
					ActorDefinidoSuperior()
					Blip.Remove($ResaltadorDeEntidad)
				end

				if Selecionas()
				then
					AparecerMenu()
					Menu.SetActiveItem($PANEL, 0)
					Text.ClearHelp()
					Text.ClearPrints()
					jump @MenuDeEdicionDelActor
				end

				if $ModoDeCompilacion == ePara_PC
				then
					while true
						int 20@ += 1
						if 20@ >= 57
						then 20@ = 48
							break
						end

						if Key.IsPressed(20@)
						then
							int 22@ = $indActor

							int 19@ = 20@
							19@ -= 48
							$indActor = 19@

							if Actor.Defined($Actor($indActor,255i))
							then 
								wait 0
								AparecerMenu()
								Menu.SetActiveItem($PANEL, 0)
								Text.ClearHelp()
								Text.ClearPrints()
								Blip.Remove($ResaltadorDeEntidad)
								jump @MenuDeEdicionDelActor
							else $indActor = 22@
							end
						end
					end
				else
					while true
						int 20@ += 1
						if 20@ >= 129
						then 20@ = 123
							break
						end

						int 19@ = 20@
						19@ -= 123

						if $numActores >= 19@
						then
							if Widget.IsPressed(20@)
							then
								int 22@ = $indActor

								$indActor = 19@

								if Actor.Defined($Actor($indActor,255i))
								then 
									wait 0
									AparecerMenu()
									Menu.SetActiveItem($PANEL, 0)
									Text.ClearHelp()
									Text.ClearPrints()
									Blip.Remove($ResaltadorDeEntidad)
									jump @MenuDeEdicionDelActor
								else $indActor = 22@
								end
							end
						end
					end
				end
			until SalirDelMenu()
			Blip.Remove($ResaltadorDeEntidad)
			AparecerMenu()
			Menu.SetActiveItem($PANEL, 3)
			jump @MenuDelActor
		end
		FrenarSeleccion()
	end
until SalirDelMenu()

Menu.SetActiveItem($PANEL, 1)
jump @MenuPrincipal
