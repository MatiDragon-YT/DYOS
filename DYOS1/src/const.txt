// temporal datas

$Hdata($INDEX,255i) = 0@ || $0  // handle
$Idata($INDEX,255i) = 0   		// int
$Fdata($INDEX,255f) = 0.0 		// float
$Sdata($INDEX,255s) = '' 		// string
$Vdata($INDEX,255v) = ""  		// long-string

// example use

if $ModoDeCompilacion == ePara_PC
then
	get_cleo_shared_var $Hdata($INDEX,255i) = ACTOR
	get_cleo_shared_var $INDEX = INDICE_DE_ACTORES
	get_cleo_shared_var 0@ = NUMERO_DE_ACTORES
else
	Android.GetVar($Hdata($INDEX,255i), ACTOR)
	Android.GetVar($INDEX, INDICE_DE_ACTORES)
	Android.GetVar(0@, NUMERO_DE_ACTORES)
end

if 0@ < 101
then
	$INDEX++
	$Hdata($INDEX,255i) = Actor.Create(0,0,0.0,0.0,0.0)
	0@++
end

if $ModoDeCompilacion == ePara_PC
then
	set_cleo_shared_var ACTOR = $Hdata($INDEX,255i)
	set_cleo_shared_var INDICE_DE_ACTORES = $INDEX
	set_cleo_shared_var NUMERO_DE_ACTORES = 0@
else
	Android.SetVar(ACTOR, $Hdata($INDEX,255i))
	Android.SetVar(INDICE_DE_ACTORES, $INDEX)
	Android.SetVar(NUMERO_DE_ACTORES, 0@)
end
